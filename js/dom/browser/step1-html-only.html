<!-- html：ルート要素 -->
<html lang="ja">

<!-- head：メタ情報 -->
<head>
  <meta charset="UTF-8" />
  <title>HTMLタグ復習ページ</title>
</head>

<!-- body：表示内容 -->
<body>

  <!-- 見出しタグ -->
  <h1>これは h1 タグです</h1>
  <h2>これは h2 タグです</h2>

  <!-- p：段落 -->
  <p>これは p タグです</p>

  <!-- strong：強調（重要） -->
  <p><strong>strong タグ</strong></p>

  <!-- em：強調（ニュアンス） -->
  <p><em>em タグ</em></p>

  <!-- br：改行 -->
  <p>改行<br />brタグ</p>

  <!-- ul / li：順序なしリスト -->
  <ul>
    <li>liタグ</li>
  </ul>

  <!-- ol：順序ありリスト -->
  <ol>
    <li>番号付き</li>
  </ol>

  <!-- a：リンク -->
  <a href="URL">リンク</a>

  <!-- img：画像 -->
  <img src="URL" alt="説明" />

  <!-- form：入力フォーム -->
  <form>
    <label>名前</label>
    <input type="text" />
    <textarea></textarea>
    <button>送信</button>
  </form>

</body>
</html>
